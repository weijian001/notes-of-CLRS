```c++
//c++代码
//注意下标从0开始，for和while循环的值需要改变
#include <iostream>
#include <vector>
using namespace std;

void print(vector<int> va);
int main()
{
    int temp[10] = {4, 5, 6, 1, 10 ,9, 8, 2, 3, 7};
    vector<int> va(temp, temp+10);
    //Insertion-Sort
    for (int i = 1; i < va.size(); i++) {
        int key = va[i];
        int j = i-1;
        while (j >= 0 && va[j] > key) {
            va[j+1] = va[j];
            j = j-1;
        }
        va[j+1] = key;
    }
    print(va);
    return 0;
}
```
